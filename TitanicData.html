"""
Retorna a contagem de ocorrncias de um determinado campo no dataframe

Argumentos:
    df_input: array multidimensional de entrada
    field: campo a ser contado
Retorna:
    list: cada posio representando um valor nico para o campo informado
"""
def fields_counter(df_input, field):
    groups = {}
    df_grouped = df_input.groupby(field)[field].count()
    
    for group in df_input[field].unique():
        groups[group] = df_grouped[group]
            
    return groups
In [84]:
"""
Retorna a porcentagem de ocorrncias de um determinado campo no dataframe

Argumentos:
    df_counter: array multidimensional de entrada com as contagens j realizadas
    df_total: array multidimensional representando o universo total de elementos
Retorna:
    list: cada posio representando um valor nico para o campo informado
"""
def fields_percent(df_counter, df_total):
    groups = {}
    
    for group in df_counter:
        groups[group] = df_counter[group] * 100 / df_total.shape[0]
            
    return groups

"""
Retorna um grfico de barras para um dicionrio informado

Argumentos:
    df_input: array multidimensional de entrada
Retorna:
    list: grfico de barras com os elementos do array de entrada ordenados pela chave
"""
def graph_bar_generator(df_input, main_title = '', x_label = '', y_label = ''):
    df_sorted = {}
    
    # Sort method: https://www.saltycrane.com/blog/2007/09/how-to-sort-python-dictionary-by-keys/
    for key in sorted(df_input.keys()):
        df_sorted[key] = df_input[key]
    
    plt.bar(range(len(df_sorted)), df_sorted.values(), align='center')
    plt.xticks(range(len(df_sorted)), list(df_sorted.keys()))
    plt.title(main_title)
    plt.xlabel(x_label)
    plt.ylabel(y_label)
    
    return plt.show()

"""
Separa os dados por faixa etria

Argumentos:
    df_input: array multidimensional de entrada
    age_range: faixa etria desejada
Retorna:
    list: data frame filtrado com a faixa etria desejada
"""
def chop_by_age_range(df_input, age_range):
    # Faixas etrias: https://pt.wikipedia.org/wiki/Predefini%C3%A7%C3%A3o:Faixas_et%C3%A1rias_do_Brasil
    def range_1():
        return "age >= 0 and age < 15"
    def range_2():
        return "age >= 15 and age < 30"
    def range_3():
        return "age >= 30 and age < 60"
    def range_4():
        return "age >= 60"
    
    # Switch/Case: https://stackoverflow.com/questions/11479816/what-is-the-python-equivalent-for-a-case-switch-statement
    options = {
        1: range_1,
        2: range_2,
        3: range_3,
        4: range_4
    }
    
    query = options[age_range]()
    return df_input.query(query)

Conclusões¶

Algumas concluses interessantes podem ser retiradas das anlises realizadas, mas que precisariam de uma maior quantidade de dados e outros fatos que talvez no possam ser explicados matematicamente, como por exemplo: qual a diferena de layout fsico do navio que levou a salvar boa parte dos passageiros de primeira classe, mas que no pode salvar tantos passageiros da terceira classe? Qual a estratgia de evacuao foi utilizada para que os grficos de sobreviventes por sexo tenham mostrado maiores ndices para os passageiros do sexo feminino?

Estatisticamente, possvel concluir que mulheres adultas da primeira classe esto entre a maioria dos sobreviventes, mas seriam necessrias mais anlises e massa de dados para que outras perguntas possam ser feitas e respondidas.

Os resultados obtidos neste trabalho devem ser considerados puramente acadmicos e anlises criteriosas devem ser feitas por especialistas para que possam ser levadas em considerao como material de pesquisa para outros fins.

Limitações¶

A quantidade de dados disponveis para a anlise no chega a representar 50% do total de passageiros do Titanic, podendo levar a resultados menos precisos ou falsos positivos. Para a anlise feita sobre as faixas etrias, por exemplo, as informaes foram ainda mais reduzidas, pois foram descartados os registros de passageiros sem idade disponvel.
